Some of the complaint types which can be categorised are internet issues

In [12]:
#Create a new feature for the complain type and categorise the complaints based on the text provided in the complaints
df['Complaint_type'] = ""
df.loc[df["Customer Complaint"].str.contains("account", case=False), ['Complaint_type']] = "Account related issues"
df.loc[((df["Customer Complaint"].str.contains("cable|connect", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Comcast cable/connection issues"
df.loc[((df["Customer Complaint"].str.contains("Pay|charg|bill|fee|pmts|credit", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Billing issues"
df.loc[((df["Customer Complaint"].str.contains("pric|rate", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Pricing issues"
df.loc[((df["Customer Complaint"].str.contains("refund", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Refund issues"
df.loc[((df["Customer Complaint"].str.contains("speed|slow|bandwidth|throttling", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Speed issues"
df.loc[((df["Customer Complaint"].str.contains("cap|data", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Data issues"
df.loc[((df["Customer Complaint"].str.contains("internet", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Internet issues"
df.loc[((df["Customer Complaint"].str.contains("Xfinity|hbo", case=False)) & (df["Complaint_type"] == "")), ['Complaint_type']] = "Comcast Xfinity/HBO service issues"
df.loc[df["Complaint_type"] == "", ['Complaint_type']] = "Comcast service related issues"

Task#6 : Which state has the highest percentage of unresolved complaints

In [22]:
df_total_tickets = df.groupby('State')[["Ticket #"]].count()
df_unresolved_tickets = df[df.Current_status == "Open"].groupby('State')[["Current_status"]].count()
df_complaints = pd.merge(df_total_tickets, df_unresolved_tickets, on="State", how='left')
df_complaints.rename(columns={"Ticket #": "Logged", "Current_status": "Unresolved"}, inplace=True)
df_complaints.fillna(0, inplace=True)
df_complaints["Pct_Unresolved"] = np.round(df_complaints.Unresolved/df_complaints.Logged * 100, 2)
df_complaints.sort_values(by="Pct_Unresolved", ascending=False).head()
